rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {

    function signedIn() {
      return request.auth != null;
    }

    function isAdmin() {
      return signedIn()
        && (
          request.auth.token.email == "asmaabd1987@gmail.com"
          || (
            exists(/databases/$(database)/documents/userProfiles/$(request.auth.uid))
            && get(/databases/$(database)/documents/userProfiles/$(request.auth.uid)).data.role == "admin"
          )
        );
    }

    function myProfile() {
      return get(/databases/$(database)/documents/userProfiles/$(request.auth.uid)).data;
    }

    function isActiveEmployee() {
      return signedIn()
        && exists(/databases/$(database)/documents/userProfiles/$(request.auth.uid))
        && myProfile().role == "employee"
        && myProfile().active == true;
    }

    function canStock() {
      return isActiveEmployee()
        && myProfile().permissions.stock == true;
    }

    function canExpenses() {
      return isActiveEmployee()
        && myProfile().permissions.expenses == true;
    }

    // Main company datasets
    match /companies/pause-plate-manager/data/{dataset} {
      allow read, write: if isAdmin();

      allow read, write: if dataset in ["products", "movements"] && canStock();
      allow read, write: if dataset == "expenses" && canExpenses();
    }

    // App metadata: Admin only. Employees do not need to modify it.
    match /companies/pause-plate-manager/meta/{document} {
      allow read, write: if isAdmin();
    }

    // User profiles:
    // - Admin manages everyone.
    // - Each user can read only their own profile.
    match /userProfiles/{uid} {
      allow read: if isAdmin() || (signedIn() && request.auth.uid == uid);
      allow create, update, delete: if isAdmin();
    }

    // Legacy user records used by the current app.
    match /users/{uid} {
      allow read: if isAdmin() || (signedIn() && request.auth.uid == uid);
      allow create, update, delete: if isAdmin();
    }

    // Audit Trail:
    // Admin can read everything.
    // Employees may only CREATE immutable audit entries for themselves.
    match /auditLogs/{logId} {
      allow read: if isAdmin();

      allow create: if isActiveEmployee()
        && request.resource.data.user.uid == request.auth.uid;

      allow update, delete: if false;
    }

    // Everything else is denied.
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
